  zksync:
    name: zkSync Foundry (Native AA)
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v5
        with:
          submodules: recursive

      - name: Install Rust
        uses: dtolnay/rust-toolchain@stable

      - name: Clone zkSync Foundry Fork
        run: |
          git clone https://github.com/matter-labs/foundry-zksync.git
          cd foundry-zksync
          cargo build --release
          echo "$PWD/target/release" >> $GITHUB_PATH

      - name: Verify Forge Version
        run: forge --version

      - name: Run Normal zkSync Tests
        run: forge test --match-path "test/ZKAA/*" --zksync -vvv --no-match-test testZKfunctionValidateTransaction

      - name: Run System Mode Tests
        run: forge test --mt testZKfunctionValidateTransaction --zksync --system-mode=true -vvv
